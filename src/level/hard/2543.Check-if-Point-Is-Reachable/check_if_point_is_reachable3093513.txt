// https://leetcode.com/problems/check-if-point-is-reachable/solutions/3093513/rust-gcd-solution/
fn gcd(x: i32, y: i32) -> i32 {
	let (mut x, mut y) = (x, y);
	while y != 0 { let t = x % y; x = y; y = t; }
	x
}

impl Solution {
	pub fn is_reachable(target_x: i32, target_y: i32) -> bool {
		let mut x = gcd(target_x, target_y);
		while x % 2 == 0 { x /= 2; }
		x == 1
	}
}
