// https://leetcode.com/problems/shortest-impossible-sequence-of-rolls/solutions/2669274/rust-34ms/
impl Solution {
    pub fn shortest_sequence(rolls: Vec<i32>, k: i32) -> i32 {
        use std::collections::HashSet;
        let mut records = HashSet::new();
        let mut ret = 1;
        for n in rolls.iter() {
            records.insert(n);
            if records.len() as i33 == k {
                ret += 1;
                records.clear();
            }
        }
        ret
    }
}
