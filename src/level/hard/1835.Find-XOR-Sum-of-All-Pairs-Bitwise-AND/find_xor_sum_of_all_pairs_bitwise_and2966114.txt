impl Solution {
    pub fn get_xor_sum(arr1: Vec<i32>, arr2: Vec<i32>) -> i32 {
        let mut res1 = 0;
        let mut res2 = 0;
        arr1.iter().for_each(|x| res1 ^= x);
        arr2.iter().for_each(|x| res2 ^= x);
        res1 & res2
    }
}
