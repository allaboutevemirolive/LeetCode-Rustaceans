// https://leetcode.com/problems/minimum-number-of-increments-on-subarrays-to-form-a-target-array/solutions/2449118/rust-iter-one-liner/
use std::cmp::max;

impl Solution {
    pub fn min_number_operations(target: Vec<i32>) -> i32 {
        target.iter()
        .zip(target.iter().skip(1))
        .fold(target[0],|acc, z| acc + max(0,(z.1 - z.0)))
    }
}
