impl Solution {
    pub fn min_number_operations(target: Vec<i32>) -> i32 {
        let mut result = 0;
        
        let mut prevNum = 0;
        
        for num in target {
            if(prevNum < num) {
                result += num - prevNum;
            }
            prevNum = num;
        }
        return result;
    }
}
