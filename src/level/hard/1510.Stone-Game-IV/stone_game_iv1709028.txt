// https://leetcode.com/problems/stone-game-iv/solutions/1709028/rust-8ms-100/
impl Solution {
    pub fn winner_square_game(n: i32) -> bool {
        let mut is_win = Vec::with_capacity((n+1) as usize);
        is_win.push(false); // Alice looses when 0 stones left
        for stones_left in 1..=n as usize {
            let mut k = 1;
            is_win.push(
                loop {
                    let step = k*k;
                    if step > stones_left {
                        break false; // did not find any move that would win
                    }
                    if !is_win[stones_left - step] {
                        break true; // winning move
                    }
                    k += 1;
                }
            );
        }
        is_win[n as usize]
    }
}
