impl Solution {
    pub fn winner_square_game(n: i32) -> bool {
        let n = n as usize;
        let mut first_win = vec![false; n + 1];
        for i in 1..=n {
            first_win[i] = (1..)
                .map(|x| x * x)
                .take_while(|&x| x <= i)
                .any(|x| !first_win[i - x]);
        }
        first_win[n]
    }
}
