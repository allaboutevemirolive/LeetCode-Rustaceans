// https://leetcode.com/problems/maximum-xor-with-an-element-from-array/solutions/1216141/c-trie-sol-with-comments/
    vector<int > res(queries.size());
    
    for(int i=0;i<queries.size();i++){
       int x=queries[i][0],m=queries[i][1];
       if(m<mn)  res[i]=-1;
       else{
           res[i]=query(x,m+1);  // inc. m by 1 to get rid of equal to m values in num
       }
       
    }
    return res;
