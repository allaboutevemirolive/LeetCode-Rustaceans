// https://leetcode.com/problems/earliest-possible-day-of-full-bloom/solutions/3356766/rust-easy-solution/
impl Solution {
    pub fn earliest_full_bloom(plant_time: Vec<i32>, grow_time: Vec<i32>) -> i32 {
        let mut p = grow_time.into_iter().zip(plant_time.into_iter()).collect::<Vec<(i32, i32)>>();
        p.sort_unstable();
        p.iter().rev().fold((0, 0), |(mut res, max_day), plants| {
            (res + plants.1, max_day.max(res + plants.0 + plants.1))
        }).1
    }
}
