// https://leetcode.com/problems/reverse-integer/solutions/2073605/rust-100/
impl Solution {
    pub fn reverse(mut x: i32) -> i32 {
        let mut signed = 1;
        let mut res = 0;
        if x < 0 {
            signed = -1;
            x = -x;
        }
        while x > 0 {
            if (res*10)/10 != res {
                return 0;
            }
            res = res*10 + x%10;
            x = x/10;
        }
        signed * res
    }
}
