// https://leetcode.com/problems/reverse-integer/solutions/1898766/rust/
fn reverse(x: i32) -> i32 {
	fn safe_reverse(mut x: i32) -> Option<i32> {
		let mut y = 0i32;
		while x != 0 {
			let i = x % 10;
			x = x / 10;
			y = y.checked_mul(10)?.checked_add(i)?;
		}
		Some(y)
	}

	safe_reverse(x).unwrap_or(0)
}
