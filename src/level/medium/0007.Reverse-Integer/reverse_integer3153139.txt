// https://leetcode.com/problems/reverse-integer/solutions/3153139/rust-string-solution/
impl Solution {
    pub fn reverse(x: i32) -> i32 {
        if x ==0 {
            return 0
        }
        let mut res:Vec<char> = x.to_string().chars().rev().skip_while(|ch| ch.eq(&'0')).collect();
        if res.last().unwrap().eq(&'-') {
            res.rotate_right(1);
        }
        res.iter().collect::<String>().parse::<i32>().unwrap_or(0)
    }
}
