// https://leetcode.com/problems/reverse-integer/solutions/2690450/rust-implementation/
impl Solution {
pub fn reverse(x: i32) -> i32 {
    let sign = if x < 0 { -1 } else { 1 };
    let mut num = x.abs();
    let mut reverse = 0;
    while num > 0 {
        let last_digit = num % 10;
        if reverse > (i32::MAX - last_digit) / 10 {
            return 0;
        }
        reverse = (reverse * 10) + last_digit;
        num = num / 10;
    }
    return sign * reverse;
}

}
