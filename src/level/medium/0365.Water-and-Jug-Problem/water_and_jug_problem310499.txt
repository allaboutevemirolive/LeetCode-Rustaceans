// https://leetcode.com/problems/water-and-jug-problem/solutions/310499/runtime-0-ms-faster-than-100-00-simple-rust-program/
fn gcd(a: i32, b: i32) -> i32 {
    if b == 0 {
        a
    } 
    else {
        gcd(b, a % b)
    }
}

impl Solution {
    pub fn can_measure_water(x: i32, y: i32, z: i32) -> bool {
        z == 0 || (x + y >= z && z % gcd(x, y) == 0)
    }
}
