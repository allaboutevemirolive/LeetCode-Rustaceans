    pub fn max_sub_array(nums: Vec<i32>) -> i32 {
        let mut max_sum = i32::MIN;
        let mut running_sum = 0;
        for num in nums {
            running_sum = std::cmp::max(num, running_sum + num);
            max_sum = std::cmp::max(max_sum, running_sum);
        }
        max_sum
    }
