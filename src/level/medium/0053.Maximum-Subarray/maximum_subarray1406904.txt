// https://leetcode.com/problems/maximum-subarray/solutions/1406904/rust-solution/
impl Solution {
    pub fn max_sub_array(nums: Vec<i32>) -> i32 {
        let mut tmp_sum = 0;

        nums.iter().fold(nums[0], move |mut max_sum, num| {
            tmp_sum += num;
            max_sum = max_sum.max(tmp_sum);
            tmp_sum = tmp_sum.max(0);

            max_sum
        })
    }
}
