// https://leetcode.com/problems/maximum-subarray/solutions/1595094/rust/
use std::cmp;
impl Solution {
    pub fn max_sub_array(nums: Vec<i32>) -> i32 {
        let mut ans = i32::MIN;
        let mut max= 0;
        if nums.len()>1{
            for (_,&v) in nums.iter().enumerate(){
                max = cmp::max(max + v, v);
                ans = cmp::max(ans, max);
            }
            return ans;
        }else{
            return nums[0]
        }
    }
}
