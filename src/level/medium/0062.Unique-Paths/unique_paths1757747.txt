    pub fn unique_paths(m: i32, n: i32) -> i32 {
        let m = m as usize;
        let n = n as usize;

        let mut dp = vec![vec![1; n]; m];

        for mi in 1_usize..m  {
            for ni in 1_usize..n {
                dp[mi][ni] = dp[mi - 1][ni] + dp[mi][ni - 1];
            }
        }

        dp[m - 1][n - 1]
    }
