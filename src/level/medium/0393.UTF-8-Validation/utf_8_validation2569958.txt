// https://leetcode.com/problems/utf-8-validation/solutions/2569958/simple-rust-runtime-0-ms-memory-usage-2-1-mb/
impl Solution {
    pub fn valid_utf8(data: Vec<i32>) -> bool {
        let mut followers = 0;
        for curr in data {
            if followers == 0 {
                match curr {
                    _ if curr & 0b1000_0000 == 0b0000_0000 => continue,
                    _ if curr & 0b1110_0000 == 0b1100_0000 => {
                        followers = 1;
                        continue;
                    }
                    _ if curr & 0b1111_0000 == 0b1110_0000 => {
                        followers = 2;
                        continue;
                    }
                    _ if curr & 0b1111_1000 == 0b1111_0000 => {
                        followers = 3;
                        continue;
                    }
                    _ => return false,
                }
            } else {
                match curr {
                    _ if curr & 0b1100_0000 == 0b1000_0000 => {
                        followers -= 1;
                        continue;
                    }
                    _ => return false,
                }
            }
        }
        match followers == 0 {
            true => true,
            false => false,
        }
    }
}
