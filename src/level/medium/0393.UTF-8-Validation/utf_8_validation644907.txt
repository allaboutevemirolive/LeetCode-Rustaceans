// https://leetcode.com/problems/utf-8-validation/solutions/644907/rust-12-lines-short-solution-0ms/
impl Solution {
    pub fn valid_utf8(data: Vec<i32>) -> bool {
        let mut left = 0; 
        for d in data.iter() {
            if left == 0 {
                if d >> 3 == 0b11110 { left = 3 }
                else if d >> 4 == 0b1110 { left = 2 }
                else if d >> 5 == 0b110 { left = 1 }
                else if d >> 7 == 0b0 { left = 0 }
                else { return false }
            } else {
                if d >> 6 != 0b10 { return false }
                left -= 1;
            }
        }
        left == 0
    }
}
