// https://leetcode.com/problems/utf-8-validation/solutions/2571280/rust-0-ms-solution/

impl Solution {
    pub fn valid_utf8(data: Vec<i32>) -> bool {
        

	let mut count = 0;

		for n in data {
			if n > 247 {
				return false;
			}
			if count == 0 {
				if n >= 240 && n <= 247 {
					count = 3;
				} else if n >= 224 && n <= 239 {
					count = 2;
				} else if n >= 192 && n <= 223 {
					count = 1;
				} else if n >= 128 && n <= 191 {
					return false;
				} else {
					if !(n >= 0 && n <= 127) {
						return false;
					}
				}
			} else {
				if n >= 128 && n <= 191 {
					count -= 1;
				} else {
					return false;
				}
			}
		}

		if count != 0 {
			return false;
		}
		return true;
		}
}
