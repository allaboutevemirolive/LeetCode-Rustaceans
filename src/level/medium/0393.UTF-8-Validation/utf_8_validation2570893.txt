// https://leetcode.com/problems/utf-8-validation/solutions/2570893/rust-100-counting-ones/
impl Solution {
    pub fn valid_utf8(data: Vec<i32>) -> bool {
        let mut len = 0;
        
        for i in 0..data.len() {
            let curr = data[i] as u8;
            if len == 0 {
                len = curr.leading_ones();
                if len == 0 {
                    continue;
                }
                
                if len > 4 || len == 1 {
                    return false;
                }
            } else {
                if curr.leading_ones() != 1 {
                    return false;
                }
            }
            len -= 1;
        }
        len == 0
    }
}
