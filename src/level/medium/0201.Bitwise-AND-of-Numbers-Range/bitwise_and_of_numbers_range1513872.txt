// https://leetcode.com/problems/bitwise-and-of-numbers-range/solutions/1513872/rust-solution/
impl Solution {
    pub fn range_bitwise_and(left: i32, right: i32) -> i32 {
        
        if right > std::i32::MAX {
            return 0;
        } else {
        
            let mut result: i32 = right;
            for i in left..right {
                result &= i;
                if result == 0 {break;}
            }
            return result;
        }
    }
}
