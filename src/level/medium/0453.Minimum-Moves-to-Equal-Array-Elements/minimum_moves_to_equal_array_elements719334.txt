// https://leetcode.com/problems/minimum-moves-to-equal-array-elements/solutions/719334/in-rust-not-idiomatic/
impl Solution {
    pub fn min_moves(nums: Vec<i32>) -> i32 {
        let mut m = nums.iter().min().unwrap();
        let mut res = 0;
        for n in &nums {
            res += n - m;
        }
        res
    }
}
