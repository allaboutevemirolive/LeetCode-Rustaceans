// https://leetcode.com/problems/minimum-moves-to-equal-array-elements/solutions/249411/simple-rust/
impl Solution {
    pub fn min_moves(nums: Vec<i32>) -> i32 {
        let mut min = nums[0];
        let mut sum = 0;
        let len = nums.len();
        for v in nums.iter() {
            sum += *v;
            if *v < min {
                min = *v;
            } 
        }
        sum - len as i32 * min
    }
}
