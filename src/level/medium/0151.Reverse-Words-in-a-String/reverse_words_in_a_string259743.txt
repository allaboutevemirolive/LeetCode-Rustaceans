// https://leetcode.com/problems/reverse-words-in-a-string/solutions/259743/simple-rust-iterator-based-solution/
fn reverse_words(s: String) -> String {
        s.split_whitespace()
         .fold("".to_string(), |acc, word| [word, acc.as_str()].join(" "))
         .trim()
         .to_string()
    }
