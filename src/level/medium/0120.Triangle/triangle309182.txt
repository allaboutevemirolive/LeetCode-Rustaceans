impl Solution {
    pub fn minimum_total(triangle: Vec<Vec<i32>>) -> i32 {
        if triangle.len() == 0 { return 0; }
        
        let mut min = triangle.last().unwrap().clone();
        for i in (0..triangle.len() - 1).rev() {
            for j in 0..triangle[i].len() {
                min[j] = min[j].min(min[j + 1]) + triangle[i][j];
            }
        }

    }
}
