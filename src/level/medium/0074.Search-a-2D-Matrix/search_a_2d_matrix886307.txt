impl Solution {
    pub fn search_matrix(matrix: Vec<Vec<i32>>, target: i32) -> bool {
         if matrix.is_empty() {
        return false;
    };
    for i in matrix.iter() {
        for &j in i.iter() {
            if j == target {
                return true;
            }
        }
    }
    false
    }
}
