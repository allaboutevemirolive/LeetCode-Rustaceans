// https://leetcode.com/problems/jump-game-ii/solutions/574818/rust-solution/
pub fn jump(nums: Vec<i32>) -> i32 {
    let mut maxReachable: usize = 0;
    let mut numSteps: i32 = 0;
    let mut maxNext: usize = 0;

    if nums.len() <= 1 {
        return 0;
    }

    for (idx, num) in nums.iter().enumerate() {
        if maxReachable + 1 >= nums.len() {
           break;
        }
        
        if idx > maxReachable { // need one more step
            numSteps = numSteps + 1;
            maxReachable = maxNext;
        }
        maxNext = std::cmp::max(maxNext, idx + (*num as usize));
    }
    numSteps
}
