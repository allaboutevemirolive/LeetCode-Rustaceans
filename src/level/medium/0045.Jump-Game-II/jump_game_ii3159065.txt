// https://leetcode.com/problems/jump-game-ii/solutions/3159065/rust-implicit-bfs-solution/
impl Solution {
    pub fn jump(nums: Vec<i32>) -> i32 {
        // This is an implicit bfs solution.
        let mut steps = 0;
        let n = nums.len();
        let (mut right, mut rightmost) = (0, 0);
        for i in 0..n - 1 {
            rightmost = rightmost.max(i as i32 + nums[i]);
            // means you visited all the items on the current level. 
            if i == right {
                // getting the queue size for the next level you are traversing
                right = rightmost as usize;
                steps += 1;  // like incrementing the level you are on
            }
        }
        steps
    }
}
