// https://leetcode.com/problems/jump-game-ii/solutions/3158196/rust-dp-solution/
impl Solution {
    pub fn jump(nums: Vec<i32>) -> i32 {
        let n = nums.len();
        let mut f = vec![0; n];

        for i in 1..n {
            for j in 0..i {
                if j + nums[j] as usize >= i {
                    f[i] = f[j] + 1;
                    break;
                }
            }
        }

        f[n - 1]
    }
}
