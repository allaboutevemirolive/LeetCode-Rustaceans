// https://leetcode.com/problems/jump-game-ii/solutions/3160441/for-rust-geeks/
impl Solution {
    pub fn jump(nums: Vec<i32>) -> i32 {
        let (n, mut next, mut jumps, mut cur) = (nums.len() - 1, 0, 0, 0);
        for i in 0..n {
            next = i32::max(next - 1, nums[i]);
            if cur == 0 {
                cur = next;
                next = 0;
                jumps += 1;
            }
            cur -= 1;
        }
        jumps
    }
}
