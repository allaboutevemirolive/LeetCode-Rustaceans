// https://leetcode.com/problems/jump-game-ii/solutions/1505308/rust-100-2-1mb/
impl Solution {
    pub fn jump(nums: Vec<i32>) -> i32 {
        let (mut furr, mut end, mut res, l1) = (0,0,0, nums.len());
        for i in 0..l1 - 1 {   
            furr = furr.max(i + nums[i] as usize);        
            if i == end { 
                res +=1;
                end = furr
            }
        }
        res as i32
    }
}
