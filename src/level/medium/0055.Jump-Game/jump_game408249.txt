// https://leetcode.com/problems/jump-game/solutions/408249/greedy-rust-100/
impl Solution {
    pub fn can_jump(nums: Vec<i32>) -> bool {
        let mut dest = nums.len() - 1;
        for i in (0..nums.len() - 1).rev() {
            if i + (nums[i] as usize) >= dest {
                dest = i;
            }
        }
        dest == 0
    }
}
