// https://leetcode.com/problems/jump-game/solutions/655359/rust-solution/
use std::cmp;

impl Solution {
    pub fn can_jump(nums: Vec<i32>) -> bool {
        let mut max = 0;
        for (i, &num) in nums.iter().enumerate() {
            if i > max {
                return false;
            }
            max = cmp::max(max, i + num as usize);
            if max > nums.len() - 1 {
                return true;
            }
        }
        true
    }
}
