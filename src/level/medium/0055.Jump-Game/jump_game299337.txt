// https://leetcode.com/problems/jump-game/solutions/299337/rust-solution/
impl Solution {
    pub fn can_jump(nums: Vec<i32>) -> bool {
        let mut step = vec![0;nums.len()];
        
        for i in 1..nums.len() {
            step[i] = step[i-1].max(nums[i-1]) - 1;
            if step[i] < 0 {
                return false;
            }
        }
        true
    }
}
