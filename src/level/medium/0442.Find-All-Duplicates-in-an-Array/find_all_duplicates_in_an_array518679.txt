// https://leetcode.com/problems/find-all-duplicates-in-an-array/solutions/518679/rust-simple-sign-flipping-solution/
impl Solution {
    pub fn find_duplicates(mut nums: Vec<i32>) -> Vec<i32> {
      let mut result = vec![];
      for i in 0..nums.len() {
        let ith = nums[i].abs();
        let index = ith as usize - 1;
        if (nums[index] < 0) {
          result.push(ith);
        } else {
          nums[index] *= -1;
        }
      }
      result
    }
}
