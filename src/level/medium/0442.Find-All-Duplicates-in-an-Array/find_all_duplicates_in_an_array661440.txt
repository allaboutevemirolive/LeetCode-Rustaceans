// https://leetcode.com/problems/find-all-duplicates-in-an-array/solutions/661440/rust-3-liner-faster-than-100/
impl Solution {
    pub fn find_duplicates(nums: Vec<i32>) -> Vec<i32> {
        let mut seq = [0].repeat(nums.len());
        nums.into_iter().for_each(|num| seq[(num-1) as usize] += 1);
        seq.into_iter().enumerate().filter(|&(x, y)| y >= 2).map(|(x, y)| (x+1) as i32).collect::<Vec<i32>>()
    }
}
