// https://leetcode.com/problems/largest-number/solutions/222005/rust-solution/
impl Solution {
    pub fn largest_number(nums: Vec<i32>) -> String {
        let mut v = nums.iter().map(|num| num.to_string()).collect::<Vec<String>>();
        v.sort_by(|a, b| format!("{}{}",b ,a).cmp(&format!("{}{}", a, b)));
        let mut res = v.join("");
        if res.trim_start_matches("0").is_empty() {
            return "0".to_string();
        }
        return res.to_string();
    }
}
