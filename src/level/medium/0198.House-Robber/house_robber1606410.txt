// https://leetcode.com/problems/house-robber/solutions/1606410/rust-dynamic-programming/
use std::cmp::max;

impl Solution {
    pub fn rob(nums: Vec<i32>) -> i32 {
        let length = nums.len();
        
        
        let mut ans: Vec<i32> = vec![0; length+1];
        ans[0] = 0;
        ans[1] = nums[0];
        
        for i in 1..length {
            ans[i+1] = max(ans[i-1] + nums[i], ans[i]);
        }
        
        ans[length]
    }
}
