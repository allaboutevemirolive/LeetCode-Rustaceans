// https://leetcode.com/problems/house-robber/solutions/2151866/rust-bottom-up/
impl Solution {
    pub fn rob(nums: Vec<i32>) -> i32 {
        let mut res = 0;
        let mut prev = 0;
        for n in nums.iter(){
            let tmp = res; 
            res = res.max(prev + n);
            prev = tmp;
        }
        res
    }
}
