// https://leetcode.com/problems/sort-colors/solutions/3167673/rust-solution/
impl Solution {
    pub fn sort_colors(nums: &mut Vec<i32>) {
        let mut i = 0;
        'op: while i < nums.len() {
            for j in i..nums.len() {
                if nums[j] < nums[i] {
                    nums.swap(i, j);
                    continue 'op;
                }
            }
            i += 1;
        }        
    }
}
