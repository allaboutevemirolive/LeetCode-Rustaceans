// https://leetcode.com/problems/sort-colors/solutions/1062704/rust-100-100/
impl Solution {
    pub fn sort_colors(nums: &mut Vec<i32>) {
      let mut cnts = [0, 0, 0];
      nums.iter().for_each(|x| { cnts[*x as usize] += 1usize; });
      nums.clear();
      cnts.iter().enumerate().for_each(|(i, cnt)| 
        nums.extend(std::iter::repeat(i as i32).take(*cnt))
      );
    }
}
