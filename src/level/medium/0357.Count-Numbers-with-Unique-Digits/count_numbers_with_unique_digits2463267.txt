// https://leetcode.com/problems/count-numbers-with-unique-digits/solutions/2463267/rust-solution/
impl Solution {
    pub fn count_numbers_with_unique_digits(n: i32) -> i32 {
        let mut result = 1;
        for i in 1..=n {
            let mut temp = 9;
            for j in 0..i-1 {
                temp *= 9 - j;
            }
            result += temp;
        }
        result
    }
}
