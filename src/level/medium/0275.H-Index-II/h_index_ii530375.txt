// https://leetcode.com/problems/h-index-ii/solutions/530375/rust-binary-search-solution/
impl Solution {
    pub fn h_index(citations: Vec<i32>) -> i32 {
        let (mut l, mut r) = (0, citations.len());
        while l < r {
            let m = (l + r) / 2;
            if citations[citations.len() - m - 1] > m as i32 {
                l = m + 1
            } else {
                r = m
            }
        }
        l as i32
    }
}
