impl Solution {
    pub fn my_pow(x: f64, n: i32) -> f64 {
        if n == 0 {
            return 1.00000;
        }
        if n == -1 {
            return 1.00000 / x;
        }
        if n % 2 != 0 {
            return Solution::my_pow(x * x, n >> 1) * x;
        }
        Solution::my_pow(x * x, n >> 1)
    }
}
