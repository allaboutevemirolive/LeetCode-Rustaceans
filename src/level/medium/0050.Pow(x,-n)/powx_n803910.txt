// https://leetcode.com/problems/powx-n/solutions/803910/0ms-rust-solution/
impl Solution {
    pub fn my_pow(x: f64, n: i32) -> f64 
{
    
    fn square(x: f64) -> f64 {
    x*x
}
             
        match n {
            -1 => 1.0 / x,
            0 => 1.0,
            1 => x,
            _ => if n%2 == 0 {
                
                square(Self::my_pow(x, n/2))
            }
            else {
                x * square(Self::my_pow(x, (n-1)/2))
            }
            }
    }
}
