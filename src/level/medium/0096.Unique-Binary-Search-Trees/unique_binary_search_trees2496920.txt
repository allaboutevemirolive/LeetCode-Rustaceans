/*
    96 - Number of unique BSTs 
    Time: O(n)
    Space: O(1)
    Note: Computes nth catalan number using formula.
*/
pub fn num_trees(n: i32) -> i32 {
    let n = n as i128;
    let a = (2..=n).fold(1, |prod, k| prod * (n + k));
    let b = (2..=n).fold(1, |prod, k| prod * k);
    (a / b) as i32
}
