// https://leetcode.com/problems/unique-binary-search-trees/solutions/408256/rust-0ms-in-5-line/
impl Solution {
    pub fn num_trees(n: i32) -> i32 {
        let n = n as usize;
        let mut memo = vec![1; n + 1];
        for i in 2..(n + 1) {
            memo[i] = (0..i).map(|j| memo[j] * memo[i - j - 1]).sum();
        }
        memo[n]
    }
}
