// https://leetcode.com/problems/unique-binary-search-trees/solutions/1614023/why-fold-in-rust-submission-cannot-work/
impl Solution {
    pub fn num_trees(n: i32) -> i32 {
        match n {
            0 => 1,
            1 => 1,
            _ => (0..n).collect::<Vec<i32>>().iter().fold(0, |acc, x| acc + num_trees(*x) * num_trees(n-1-x)),
        }
    }
}
