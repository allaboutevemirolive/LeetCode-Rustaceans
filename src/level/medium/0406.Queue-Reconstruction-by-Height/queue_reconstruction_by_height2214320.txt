// https://leetcode.com/problems/queue-reconstruction-by-height/solutions/2214320/rust-n-2-binaryindexedtree-solutions/
impl Solution {
    pub fn reconstruct_queue(mut people: Vec<Vec<i32>>) -> Vec<Vec<i32>> {
        people.sort_unstable_by_key(|person| (-person[0], person[1]));
        let mut result = vec![];

        for person in people {
            let num_taller = person[1];
            result.insert(num_taller as usize, person);
        }

        result
    }
}
