// https://leetcode.com/problems/queue-reconstruction-by-height/solutions/2218124/rust-solution-greedy-approach/
impl Solution {
    pub fn reconstruct_queue(mut people: Vec<Vec<i32>>) -> Vec<Vec<i32>> {
        let mut res = Vec::with_capacity(2000);
        people.sort_unstable_by(|a, b| if a[0] == b[0] {a[1].cmp(&b[1])} else {b[0].cmp(&a[0])});
        for i in &people {
            res.insert(i[1] as usize, i.clone());
        }
        res
    }
}
