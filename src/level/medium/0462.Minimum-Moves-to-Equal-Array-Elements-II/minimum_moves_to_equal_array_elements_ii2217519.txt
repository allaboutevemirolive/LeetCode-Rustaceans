impl Solution {
    pub fn min_moves2(mut nums: Vec<i32>) -> i32 {
        nums.sort_unstable();
        let median = nums[nums.len()/2];
        nums.into_iter().map(|num| ((num - median) as i64).abs()).sum::<i64>() as _
    }
}
