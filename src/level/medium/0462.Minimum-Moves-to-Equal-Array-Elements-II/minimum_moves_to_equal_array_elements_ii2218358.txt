// https://leetcode.com/problems/minimum-moves-to-equal-array-elements-ii/solutions/2218358/rust-diff-with-median-o-n-avg/
impl Solution {
    pub fn min_moves2(mut nums: Vec<i32>) -> i32 {
        let mid = nums.len() / 2;
        
		// quickselect algorithm for finding a median
        nums.select_nth_unstable(mid);
        
        let median = nums[mid];
        
        nums.iter().fold(0i32, |moves, num| moves + (median - num).abs())
    }
}
