// https://leetcode.com/problems/minimum-moves-to-equal-array-elements-ii/solutions/1218293/rust-python3-sort-and-make-all-values-equal-with-center-value/
class Solution:
    def minMoves2(self, nums: List[int]) -> int:
        nums.sort()
        
        if len(nums) % 2 == 0:
            rc = len(nums)//2
            lc = rc-1
            avg_median = (nums[rc]+nums[lc])/2
        else:
            avg_median = nums[len(nums)//2]

        return int(sum([abs(i-avg_median) for i in nums]))
