// https://leetcode.com/problems/next-permutation/solutions/286862/rust-solution/
impl Solution {
    pub fn next_permutation(nums: &mut Vec<i32>) {
        let s = nums.as_mut_slice();
        let mut i = s.len() - 1;
        while i >= 1 && s[i - 1] >= s[i] {
            i -= 1;
        }
        if i >= 1 {
            let mut j = s.len() - 1;
            while s[j] <= s[i - 1] {
                j -= 1;
            }
            s.swap(i - 1, j);
        }
        s[i..].reverse();
    }
}
