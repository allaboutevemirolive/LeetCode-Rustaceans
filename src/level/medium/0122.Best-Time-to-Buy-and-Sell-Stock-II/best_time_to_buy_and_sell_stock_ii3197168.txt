// https://leetcode.com/problems/best-time-to-buy-and-sell-stock-ii/solutions/3197168/rust-simple-one-pass-solution/
impl Solution {
    pub fn max_profit(prices: Vec<i32>) -> i32 {
        let mut res: i32 = 0;
        for i in 1..prices.len() {
            if prices[i] > prices[i - 1] {
                res += prices[i] - prices[i - 1];
            }
        }        
        return res;
    }
}
