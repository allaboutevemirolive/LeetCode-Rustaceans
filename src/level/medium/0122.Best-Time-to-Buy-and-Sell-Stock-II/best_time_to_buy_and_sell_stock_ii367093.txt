// https://leetcode.com/problems/best-time-to-buy-and-sell-stock-ii/solutions/367093/rust-solution/
impl Solution {
    pub fn max_profit(prices: Vec<i32>) -> i32 {
        if prices.is_empty() {
            return 0;
        }
        prices
            .iter()
            .zip(prices[1..].iter())
            .fold(0, |acc, (a, b)| if a < b { b - a + acc } else { acc })
    }
}
