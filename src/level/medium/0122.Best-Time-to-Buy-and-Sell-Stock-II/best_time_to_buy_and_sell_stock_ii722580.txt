// https://leetcode.com/problems/best-time-to-buy-and-sell-stock-ii/solutions/722580/rust-solutions/
impl Solution {
    pub fn max_profit(prices: Vec<i32>) -> i32 {
        let mut profit = 0;
        for i in 1..prices.len() {
            profit += 0.max(prices[i] - prices[i - 1]);
        }
        profit
    }
}
