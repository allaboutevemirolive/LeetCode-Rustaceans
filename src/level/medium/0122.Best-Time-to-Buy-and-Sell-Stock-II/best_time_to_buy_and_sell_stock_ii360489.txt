// https://leetcode.com/problems/best-time-to-buy-and-sell-stock-ii/solutions/360489/rust-folding-solution/
impl Solution {
    pub fn max_profit(prices: Vec<i32>) -> i32 {
        
        prices.iter().fold( (0, None), |(ret, last), e| {
            (
             ret +e - last.map(|v| if e > v {v} else {e}).unwrap_or(e), 
             Some(e)
            )
        }).0
    }
}
