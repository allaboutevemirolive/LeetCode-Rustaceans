impl Solution {
    pub fn max_profit(prices: Vec<i32>) -> i32 {
        let mut ans: i32 = 0;
        for i in 1..prices.len() {
            if prices[i-1] < prices[i] {
                ans += prices[i] - prices[i-1];
            }
        }
        
        ans
    }
}
