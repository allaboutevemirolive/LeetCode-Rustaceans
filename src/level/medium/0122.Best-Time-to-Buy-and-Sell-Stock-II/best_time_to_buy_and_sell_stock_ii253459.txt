// https://leetcode.com/problems/best-time-to-buy-and-sell-stock-ii/solutions/253459/rust-solution-with-0ms/
pub fn solution(prices: Vec<i32>) -> i32 {
    let length = prices.len();

    if length <= 1 {
      return 0;
    }

    let mut ret = 0;
    for i in 1..length {
      let tmp = prices[i] - prices[i - 1];
      if tmp > 0 { ret += tmp; }
    }

    ret
  }
