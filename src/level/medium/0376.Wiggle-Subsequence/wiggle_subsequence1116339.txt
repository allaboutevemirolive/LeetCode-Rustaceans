// https://leetcode.com/problems/wiggle-subsequence/solutions/1116339/rust/
impl Solution {
    pub fn wiggle_max_length(nums: Vec<i32>) -> i32 {
        nums.windows(2)
            .fold((1, 0), |(count, pn), n| {
                let diff = n[1] - n[0];
                if diff == 0 || (pn as i32).signum() == diff.signum() {
                    (count, pn)
                } else {
                    (count + 1, diff)
                }
            })
            .0
    }
}
