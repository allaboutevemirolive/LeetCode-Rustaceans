// https://leetcode.com/problems/rotate-array/solutions/1604496/rust-12-ms/
impl Solution {    
    pub fn rotate(nums: &mut Vec<i32>, k: i32) {
        let len = nums.len();
        let d = k as usize % len;

        if d == 0 {
            return;
        }
        let mut v = nums.split_off(len - d);
        v.append(nums);
        nums.append(&mut v);
    }
}
