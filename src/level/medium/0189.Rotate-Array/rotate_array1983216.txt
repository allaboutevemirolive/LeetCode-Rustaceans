// https://leetcode.com/problems/rotate-array/solutions/1983216/rust-solution/
fn rotate(nums: &mut Vec<i32>, k: i32) {
    if (k == 0) {
        return;
    } else {
        let k = k as usize % nums.len();
        if k == 0 {
            return;
        }
    }

    let start_from = nums.len() - k;
    let mut p0 = nums
        .drain(0..start_from)
        .collect::<Vec<i32>>();

    nums.append(&mut p0);
}
