// https://leetcode.com/problems/longest-increasing-subsequence/solutions/2129898/rust-dp/
impl Solution {
    pub fn length_of_lis(nums: Vec<i32>) -> i32 {
        let n = nums.len();
        let mut res = 1;
        let mut dp = vec![1;n];
        for i in 0..n{
            for j in 0..=i{
                if nums[i] > nums[j]{
                    dp[i] = dp[i].max(dp[j]+1);
                }
            }
            res = res.max(dp[i]);
        }
        res
    }
}
