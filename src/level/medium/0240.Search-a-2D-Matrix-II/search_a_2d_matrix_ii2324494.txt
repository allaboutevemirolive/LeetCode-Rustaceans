// https://leetcode.com/problems/search-a-2d-matrix-ii/solutions/2324494/rust-two-answers-binary-search-with-boundary-walking-the-grid/
impl Solution {
    pub fn search_matrix(matrix: Vec<Vec<i32>>, target: i32) -> bool {
        let mut rpi = matrix[0].len();  // rightmost-possible index
        for row in matrix {
            let maybe = row[ 0 .. rpi ].binary_search( &target );  // *ex*clusive range
            match maybe {
                Result::Ok(_) => return true,
                Result::Err(i) => rpi = i,
            }
        }
        false
    }
}
