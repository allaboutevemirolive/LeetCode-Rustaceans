// https://leetcode.com/problems/search-a-2d-matrix-ii/solutions/1504484/rust-two-solutions/

pub fn search_matrix(matrix: Vec<Vec<i32>>, target: i32) -> bool {
    for row in matrix.iter() {
        if row[0] > target {
            break;
        }

        if row[row.len() - 1] < target {
            continue;
        }

        if row.binary_search(&target).is_ok() {
            return true;
        }
    }

    false
}
