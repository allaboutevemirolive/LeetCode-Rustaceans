// https://leetcode.com/problems/swap-nodes-in-pairs/solutions/2878674/rust/
impl Solution
{
    pub fn swap_pairs(mut head: Option<Box<ListNode>>) -> Option<Box<ListNode>>
    {
        return Self::swap(head);
    }
    
    fn swap(mut curr: Option<Box<ListNode>>) -> Option<Box<ListNode>>
    {
        if let Some(mut one) = curr
        {
            if let Some(mut two) = one.next
            {
                let next = two.next;
                one.next = Self::swap(next);
                
                two.next = Some(one);
                return Some(two);
            }
            return Some(one);
        }
        return None;
    }
}
