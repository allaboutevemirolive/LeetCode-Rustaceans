// https://leetcode.com/problems/swap-nodes-in-pairs/solutions/3295813/my-rust-solution/
impl Solution {
    pub fn swap_pairs(mut head: Option<Box<ListNode>>) -> Option<Box<ListNode>> {
        let mut change = head.as_mut();
        while let Some(node) = change {
            if let Some(next) = node.next.as_mut() {
                std::mem::swap(&mut node.val, &mut next.val);
            } else { 
                break
            }
            change = node.next.as_mut().unwrap().next.as_mut();
        }
        head
    }
}
