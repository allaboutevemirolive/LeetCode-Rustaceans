// https://leetcode.com/problems/132-pattern/solutions/1190584/rust/
impl Solution {
    pub fn find132pattern(nums: Vec<i32>) -> bool {
      let mut res = false;  
      let mut minn = std::i32::MAX;
      for i in 0..(nums.len() -1) {
        minn = minn.min(nums[i]);
        for j in (i+1)..nums.len() {
          if nums[j] < nums[i] && minn < nums[j] {
            res = true
          }
        }
      }
      res
    }
}
