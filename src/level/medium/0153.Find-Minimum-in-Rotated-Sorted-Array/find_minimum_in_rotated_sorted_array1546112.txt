    pub fn find_min(nums: Vec<i32>) -> i32 {
        let mut l = 0;
        let mut r = nums.len() - 1;
        while nums[l] > nums[r] {
            let mid = (l + r) / 2;
            if nums[mid] >= nums[l] {
                l = mid + 1;
            } else {
                r = mid;
            }
