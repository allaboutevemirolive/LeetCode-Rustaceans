// https://leetcode.com/problems/sort-characters-by-frequency/solutions/2871299/rust-in-place-sorting-unsafe/
impl Solution {
    pub fn frequency_sort(mut s: String) -> String {
        let mut count = [0; b'z' as usize + 1];
        for b in s.bytes() {
            count[b as usize] += 1;
        }
        let bytes = unsafe { s.as_bytes_mut() };
        bytes.sort_unstable_by_key(|&b| (-count[b as usize], b));
        s
    }
}
