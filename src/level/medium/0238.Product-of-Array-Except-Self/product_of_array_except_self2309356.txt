// https://leetcode.com/problems/product-of-array-except-self/solutions/2309356/simple-rust-solution/
impl Solution {
    pub fn product_except_self(nums: Vec<i32>) -> Vec<i32> {
        
        let len = nums.len();
        
		// Compute Prefix Products
        let mut prefix_products = vec![1; len];
        let mut cumulative_prefix_product = 1;
        for i in 0..len {
            prefix_products[i] = cumulative_prefix_product;
            cumulative_prefix_product *= nums[i];
        } 
        
		// Compute Suffix Products
        let mut suffix_products = vec![1; len];
        let mut cumulative_suffix_product = 1;
        for i in (0..len).rev() {
            suffix_products[i] = cumulative_suffix_product;
            cumulative_suffix_product *= nums[i]
        }
        
        let mut results = vec![1;len];
        for i in 0..len {
            results[i] = prefix_products[i] * suffix_products[i];
        }
        
        results
    }
}
