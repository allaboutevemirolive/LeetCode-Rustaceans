// https://leetcode.com/problems/gray-code/solutions/3354453/rust-elixir-playing-with-bits/
impl Solution {
    pub fn gray_code(n: i32) -> Vec<i32> {
        let mut ans = Vec::with_capacity(2usize.pow(n as u32));
        ans.push(0);
        let mut val = 0;
        for i in 1..ans.capacity() as i32 {
            val ^= i & -i;
            ans.push(val);
        }
        ans
    }
}
