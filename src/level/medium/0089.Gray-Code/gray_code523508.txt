// https://leetcode.com/problems/gray-code/solutions/523508/rust-xor-0ms-solution/
impl Solution {
    pub fn gray_code(n: i32) -> Vec<i32> {
        let length = 2_i32.pow(n as u32);
        let mut result = vec![0; length as usize];
        
        for i in 0..length {
            result[i as usize] = i ^ (i >> 1);
        }
        return result;
    }
}
