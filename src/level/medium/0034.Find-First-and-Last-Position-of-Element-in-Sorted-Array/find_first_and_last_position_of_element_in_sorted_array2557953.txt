// https://leetcode.com/problems/find-first-and-last-position-of-element-in-sorted-array/solutions/2557953/rust-solution/
pub fn search_range(nums: Vec<i32>, target: i32) -> Vec<i32> {
        match nums.len() {
            _ if nums.contains(&target) => vec![nums.iter().position(|&x| x == target).unwrap() as i32, (nums.len()-1-nums.iter().rev().position(|&x| x == target).unwrap()) as i32],
            _ => vec![-1, -1]
        }
    }
