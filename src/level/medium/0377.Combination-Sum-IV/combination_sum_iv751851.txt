// https://leetcode.com/problems/combination-sum-iv/solutions/751851/rust-cheapest-best/
impl Solution {
    pub fn combination_sum4(mut nums: Vec<i32>, target: i32) -> i32 {
        let target = target as usize;
        let mut dp = vec![1];
        dp.extend(vec![0; target]);
        for i in 1..=target {
            for n in nums.iter().filter(|n| **n <= i as i32) {
                dp[i] += dp[i - *n as usize]
            }
        }
        dp[target]
    }
}
