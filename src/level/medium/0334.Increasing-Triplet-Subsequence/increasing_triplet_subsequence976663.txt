// https://leetcode.com/problems/increasing-triplet-subsequence/solutions/976663/rust-solution/
use std::i32::MAX;

impl Solution {
    pub fn increasing_triplet(nums: Vec<i32>) -> bool {
        let (mut s, mut l) = (MAX, MAX);
        for n in nums {
            if n < s {
                s = n;
            } else if n > s && n < l {
                l = n;
            } else if n > s && n > l {
                return true;
            }
        }
        false
    }
}
