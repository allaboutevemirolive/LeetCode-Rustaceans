// https://leetcode.com/problems/increasing-triplet-subsequence/solutions/2689661/rust-monotonic-stack-with-comments/
const LEN: usize = 2;

impl Solution {
    pub fn increasing_triplet(nums: Vec<i32>) -> bool {
        let mut stack = Vec::<i32>::with_capacity(LEN);
        for num in nums {
            match stack.partition_point(|e| *e < num) {
                LEN => return true,
                i if i == stack.len() => stack.push(num),
                i => stack[i] = num,
            }
        }
        false
    }
}
