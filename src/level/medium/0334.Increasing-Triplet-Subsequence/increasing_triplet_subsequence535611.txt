// https://leetcode.com/problems/increasing-triplet-subsequence/solutions/535611/rust-0ms-solution/
impl Solution {
    pub fn increasing_triplet(nums: Vec<i32>) -> bool {
        let mut small = i32::max_value();
        let mut large = small;
        
        for i in nums {
            if i <= small {
                small = i;
            } else if i <= large {
                large = i;
            } else {
                return true;
            }
        }
        
        false
    }
}
