// https://leetcode.com/problems/best-time-to-buy-and-sell-stock-with-cooldown/solutions/1522047/rust-solution/
impl Solution {
    pub fn max_profit(prices: Vec<i32>) -> i32 {
        prices
            .into_iter()
            .fold((0, 0, i32::MIN), |(a, b, c), x| {
                (a.max(c + x), b.max(a), c.max(b - x))
            })
            .0
    }
}
