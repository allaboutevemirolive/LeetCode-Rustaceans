// https://leetcode.com/problems/best-time-to-buy-and-sell-stock-with-cooldown/solutions/2941574/rust-fold-on-i32-3/
impl Solution {
pub fn max_profit(prices: Vec<i32>) -> i32 {
    let [b, _, c] = prices.iter().fold([0, i32::MIN, 0], |[b, h, c], &p|
        [b.max(c), h.max(b-p), h+p]
    );
    b.max(c)
}
}
