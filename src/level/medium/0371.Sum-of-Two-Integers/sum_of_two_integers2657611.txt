// https://leetcode.com/problems/sum-of-two-integers/solutions/2657611/simple-solution-in-rust/
impl Solution {
    pub fn get_sum(a: i32, b: i32) -> i32 {
        let mut x = a.clone();
        let mut y = b.clone();
        while y != 0 {
            let carry = (&x & &y) << 1;
            x = &x ^ &y;
            y = carry;
        }
        return x;
    }
}
