// https://leetcode.com/problems/sum-of-two-integers/solutions/539601/rust-solution-0-ms/
impl Solution {
    pub fn get_sum(a: i32, b: i32) -> i32 {
        fn sum_helper(a: i32, b: i32) -> i32 {
            if b == 0 {
                a
            } else {
                sum_helper(a ^ b , (a & b) << 1)
            }
        }
        
        sum_helper(a, b)
    }
}
