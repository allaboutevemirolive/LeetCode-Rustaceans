// https://leetcode.com/problems/sum-of-two-integers/solutions/3421592/rust-concise-solution-using-binary-adder-circuit-concept/
impl Solution {
    pub fn get_sum(a: i32, b: i32) -> i32 {
        if b == 0 {
            a
        } else {
            Solution::get_sum(a ^ b, (a & b) << 1)
        }
    }
}
