// https://leetcode.com/problems/kth-smallest-element-in-a-sorted-matrix/solutions/410156/rust-binary-heap-solution/
use std::collections::BinaryHeap;

impl Solution {
    pub fn kth_smallest(matrix: Vec<Vec<i32>>, k: i32) -> i32 {
        let mut heap = BinaryHeap::new();
        
        for row in matrix {
            for item in row {
                heap.push(item);
                if heap.len() > k as usize {
                    heap.pop();
                }
            }
        }
        heap.peek().unwrap().to_owned()
    }
}
