// https://leetcode.com/problems/kth-smallest-element-in-a-sorted-matrix/solutions/2367470/c-rust-min-heap-easy/
class Solution {
public:
	int kthSmallest(vector<vector<int>>& matrix, int k) {
		priority_queue<int,vector<int>,greater<int>> q;

		for(int i=0;i<matrix.size();i++){
			for(int j=0;j<matrix.size();j++){q.push(matrix[i][j]);}
		}

		for(int i=0;i<k-1;i++)q.pop();
		return q.top();
	}
};
