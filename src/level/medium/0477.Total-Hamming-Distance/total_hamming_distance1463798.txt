// https://leetcode.com/problems/total-hamming-distance/solutions/1463798/beautiful-and-elegant-rust-solution/
impl Solution {
    pub fn total_hamming_distance(nums: Vec<i32>) -> i32 {
        let n = nums.len() as i32;
        (0..=31)
            .map(|k| {
                let c1 = nums.iter().map(|x| (x >> k) & 1).sum::<i32>();
                (n - c1) * c1
            })
            .sum()
    }
}
