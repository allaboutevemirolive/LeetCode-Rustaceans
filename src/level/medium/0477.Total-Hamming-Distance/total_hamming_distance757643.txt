// https://leetcode.com/problems/total-hamming-distance/solutions/757643/rust-cheapest-best/
impl Solution {
    pub fn total_hamming_distance(nums: Vec<i32>) -> i32 {
        (0..30).fold(0, |sum, i| {
            let (a, b) = nums.iter().fold((0, 0), |(a, b), n| match (1 << i) & n {
                    0 => (a + 1, b),
                    _ => (a, b + 1),
            });
            sum + a * b
        })
    }
}
