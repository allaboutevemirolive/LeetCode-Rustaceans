// https://leetcode.com/problems/total-hamming-distance/solutions/848991/rust-brute-force-solution/
impl Solution {
    pub fn total_hamming_distance(nums: Vec<i32>) -> i32 {
        let (mut i, mut res) = (0, 0);
        while i < nums.len() {
            let mut j = i+1;
            while j < nums.len() {
                res += (nums[i]^nums[j]).count_ones() as i32;
                j += 1;
            }
            i += 1;
        }
        res
    }
}
