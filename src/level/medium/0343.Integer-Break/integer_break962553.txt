use std::cmp;

impl Solution {
    pub fn integer_break(n: i32) -> i32 {
        if n == 2 { return 1; }
        if n == 3 { return 2; }
        return cmp::max(
            3_i32.pow((n / 3) as u32) * (n % 3),
            3_i32.pow((n / 3 - 1) as u32) * (n % 3 + 3)
        );
    }
}
