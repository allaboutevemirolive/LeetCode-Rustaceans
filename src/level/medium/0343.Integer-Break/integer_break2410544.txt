// https://leetcode.com/problems/integer-break/solutions/2410544/rust-dp/
use std::cmp::max;

impl Solution {
    pub fn integer_break(n: i32) -> i32 {
        let n = n as usize;
        let mut dp = vec![0; n+1];
        dp[1] = 1;
        for i in 2..n+1 {
            for j in 1..i {
                dp[i] = max(dp[i], max(j, dp[j]) * max(i- j, dp[i- j]));
            }
        }
        dp[n] as i32
    }
}
