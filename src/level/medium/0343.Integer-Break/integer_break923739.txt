// https://leetcode.com/problems/integer-break/solutions/923739/rust-0ms/
impl Solution {
    pub fn integer_break(n: i32) -> i32 {
        let n = n as usize;
        
        let mut f = vec![0; n + 1];
        f[2] = 1;
        
        for i in 3..n+1 {
            for j in 1..i {
                f[i] = f[i].max(f[j].max(j) * (i - j));
            }
        }
        
        f[n] as i32
    }
}
